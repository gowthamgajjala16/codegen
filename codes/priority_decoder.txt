module priority_decoder_3to8 (
    input [2:0] in,
    output reg [7:0] out
);
    always @(*) begin
        out = 8'b00000000;
        casez (in)
            3'b1??: out = 8'b10000000;
            3'b01?: out = 8'b01000000;
            3'b001: out = 8'b00100000;
            3'b000: out = 8'b00010000;
            default: out = 8'b00000000;
        endcase
    end
endmodule
